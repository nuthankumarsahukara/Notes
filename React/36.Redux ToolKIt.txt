Store.jsx

import {configureStore, createSlice} from '@reduxjs/toolkit';

const data ={
    balance : 0,
    fullName : "",
    mobile :null,
}
const userSlice=createSlice({
    name: 'user',
    initialState :data,
    reducers: {
        updateMobile : (state,action) => {
            state.mobile=action.payload;
            },
        updateName : (state,action) => {
            state.fullName=action.payload;
        },
        withdraw : (state,action) => {
            state.balance -= action.payload;
        },
        deposit : (state,action) => {
            state.balance += +action.payload;
        }
    }
});

export const { updateMobile, updateName, withdraw, deposit } = userSlice.actions;

const store = configureStore({
    reducer : {
        user:userSlice.reducer
    }
})

export default store;

App.jsx

import React, { useState } from 'react';
import { useSelector, useDispatch } from 'react-redux';
import { deposit, withdraw, updateName, updateMobile } from './store'; 

function App() {
  // Accessing user state from Redux
  const { balance, fullName, mobile } = useSelector((state) => state.user);
  const dispatch = useDispatch();

  // Local component state
  const [amount, setAmount] = useState('');
  const [name, setName] = useState('');
  const [phone, setPhone] = useState('');

  // Dispatching actions
  const handleDeposit = () => {
    if (!isNaN(amount) && amount !== '') {
      dispatch(deposit(Number(amount)));
      setAmount('');
    }
  };

  const handleWithdraw = () => {
    if (!isNaN(amount) && amount !== '') {
      dispatch(withdraw(Number(amount)));
      setAmount('');
    }
  };

  const handleUpdateName = () => {
    if (name.trim()) {
      dispatch(updateName(name));
      setName('');
    }
  };

  const handleUpdateMobile = () => {
    if (phone.trim()) {
      dispatch(updateMobile(phone));
      setPhone('');
    }
  };

  return (
    <div style={{ padding: "20px", fontFamily: "Arial" }}>
      <h1>üè¶ Bank Account</h1>
      <h2>Balance: ‚Çπ{balance}</h2>
      <h2>Full Name: {fullName || 'Not Set'}</h2>
      <h2>Mobile: {mobile || 'Not Set'}</h2>

      <div style={{ marginTop: "20px" }}>
        <input
          type="number"
          placeholder="Enter Amount"
          value={amount}
          onChange={(e) => setAmount(e.target.value)}
        />&nbsp;
        <button onClick={handleDeposit}>Deposit</button>&nbsp;
        <button onClick={handleWithdraw}>Withdraw</button>
      </div>

      <div style={{ marginTop: "20px" }}>
        <input
          type="text"
          placeholder="Enter Full Name"
          value={name}
          onChange={(e) => setName(e.target.value)}
        /> &nbsp;
        <button onClick={handleUpdateName}>Update Name</button>
      </div>

      <div style={{ marginTop: "20px" }}>
        <input
          type="text"
          placeholder="Enter Mobile"
          value={phone}
          onChange={(e) => setPhone(e.target.value)}
        /> &nbsp;
        <button onClick={handleUpdateMobile}>Update Mobile</button>
      </div>
    </div>
  );
}

export default App;
