What is a Portal?
A Portal allows rendering a component outside its parent hierarchy in the DOM.

Syntax:
	ReactDOM.createPortal(child, container)
	
	child: The React element to render.
	container: A DOM node (e.g. document.getElementById('portal-root')).

Use Cases:
-->Modals
-->Tooltips
-->Dropdowns
-->Popovers
-->Context menus

--->Portals donâ€™t break React hierarchy, but render outside the DOM hierarchy.
--->Event bubbling still works through portals.
--->Ideal when you need to avoid CSS stacking/overflow issues.


Example:

In public/index.html:

<div id="root"></div>
<div id="portal-root"></div>

In Component:

import React from 'react';
import ReactDOM from 'react-dom';

function Modal({ children }) {
  return ReactDOM.createPortal(
    <div className="modal">{children}</div>,
    document.getElementById('portal-root')
  );
}

function App() {
  return (
    <>
      <Modal>Nuthan</Modal>
    </>
  );
}

export default App;

In Props:

Modal Example:-

import './App.css';
import ReactDOM from 'react-dom';
import 'bootstrap/dist/css/bootstrap.min.css';
const { useState, default: React } = require("react");

function Modal(props){
    return(
        ReactDOM.createPortal(
        <div className='model-overlay'>
            <div className='content'>
                <h1>This is Model heading</h1>
                <p>This is Model content</p>
                <button className='btn btn-danger' onClick={props.close}>Close</button>
            </div>
        </div>,document.body
        )
    )
}

function App(){
    const[showModal,setShowModal]=useState(false);
    
    function toggleModal(){
        setShowModal(!showModal)
    }
    
    return(
        <>
            <div className="app">
            <button className='btn btn-primary m-4' onClick={toggleModal}>Open Model</button>
            {showModal && <Modal close={toggleModal}/>}
            </div>
        </>
    )
}

export default App;
