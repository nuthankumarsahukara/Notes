A custom hook in React is a JavaScript function whose name starts with use and that can call other hooks. Custom hooks allow you to extract component logic into reusable functions.

Why Use Custom Hooks?
1.Reuse logic across multiple components.
2.Make components cleaner and more readable.
3.Encapsulate side effects or state management.

Basic Syntax
import { useState, useEffect } from 'react';

function useCustomHook() {
  const [data, setData] = useState(null);

  useEffect(() => {
    // logic here
    setData("Hello from custom hook!");
  }, []);

  return data;
}

Example:-

App.js

import useGetApiData from './util';

function App(){
  let product= useGetApiData('https://fakestoreapi.com/products/15')
  return(
    <>
      <h1>Custom Hook Example </h1>
      <div>{Object.keys(product)}</div>
      <div>{product.price}</div>
      <div>{product.title}</div>
    </>
  )
}
export default App;

util.js

import { useState,useEffect } from "react";
function useGetApiData(url){
    let [data,setData]=useState({});
    useEffect(() => {
        getDetails();
    }, []);

async function getDetails(){
        let res= await fetch(url);
        let apidata=await res.json();
        setData(apidata);
    }

    return data;
}

export default useGetApiData;
